\section{Test Strategy}
\begin{itemize}[leftmargin=*]
  \item Validate requirements from SRS (FR and NFR IDs).
  \item Combine manual exploratory testing with checklist-based validation.
  \item Prioritize interaction, accessibility, and performance paths.
\end{itemize}

\section{Scope of Testing}
\subsection{In Scope}
\begin{itemize}[leftmargin=*]
  \item Button activation by pointer and keyboard.
  \item Night transition correctness.
  \item Aurora animation behavior and reduced-motion behavior.
  \item Responsive layout at representative viewport sizes.
\end{itemize}

\subsection{Out of Scope}
\begin{itemize}[leftmargin=*]
  \item Backend or API testing (none in v1).
  \item Legacy browser compatibility testing outside target matrix.
\end{itemize}

\section{Test Environment}
\begin{itemize}[leftmargin=*]
  \item OS: Windows baseline workstation.
  \item Browsers: latest Chrome, Edge, Firefox.
  \item Tools: Browser DevTools, Lighthouse, manual keyboard test.
\end{itemize}

\section{Entry and Exit Criteria}
\subsection{Entry}
\begin{itemize}[leftmargin=*]
  \item FR-01 to FR-06 implementation complete.
  \item Build deployable in local browser.
\end{itemize}

\subsection{Exit}
\begin{itemize}[leftmargin=*]
  \item All critical and high defects resolved or explicitly deferred.
  \item All must-have test cases pass.
  \item RTM updated with execution status.
\end{itemize}

\section{Defect Management}
\begin{itemize}[leftmargin=*]
  \item Severity levels: Critical, High, Medium, Low.
  \item Critical/High must be resolved before MVP release.
\end{itemize}
